3. Create `src/lib/format.ts` — currency helpers. productVariants.price is a Drizzle decimal returned as a string (per PITFALLS — decimals round-trip as strings in mysql2). The helper must accept both:
```typescript
export function formatMYR(price: string | number): string {
  const n = typeof price === "string" ? parseFloat(price) : price;
  if (!Number.isFinite(n)) return "RM 0.00";
  return `RM ${n.toFixed(2)}`;
}

/**
 * Render the price of a product's variants as either a single price
 * (all variants equal) or a range "RM 18.00 - RM 45.00" (sorted ascending).
 */
export function priceRangeMYR(variants: Array<{ price: string | number }>): string {
  if (!variants.length) return "RM 0.00";
  const prices = variants
    .map(v => (typeof v.price === "string" ? parseFloat(v.price) : v.price))
    .filter(n => Number.isFinite(n))
    .sort((a, b) => a - b);
  if (prices.length === 0) return "RM 0.00";
  if (prices[0] === prices[prices.length - 1]) return formatMYR(prices[0]);
  return `${formatMYR(prices[0])} - ${formatMYR(prices[prices.length - 1])}`;
}
```

4. Create `src/lib/catalog.ts` — server-only product read helpers. These REPLACE ad-hoc Drizzle queries in customer-facing pages. Every helper filters `isActive = true` per D2-08, enforcing the active-only rule in one place:
```typescript
import "server-only";
import { db } from "@/lib/db";
import { products, productVariants, categories } from "@/lib/db/schema";
import { and, eq, desc } from "drizzle-orm";

const ACTIVE_WITH = {
  with: { variants: true, category: true },
} as const;

export async function getActiveProducts() {
  return db.query.products.findMany({
    where: eq(products.isActive, true),
    ...ACTIVE_WITH,
    orderBy: [desc(products.createdAt)],
  });
}

export async function getActiveFeaturedProducts(limit = 4) {
  return db.query.products.findMany({
    where: and(eq(products.isActive, true), eq(products.isFeatured, true)),
    ...ACTIVE_WITH,
    orderBy: [desc(products.createdAt)],
    limit,
  });
}

export async function getActiveProductBySlug(slug: string) {
  return db.query.products.findFirst({
    where: and(eq(products.slug, slug), eq(products.isActive, true)),
    ...ACTIVE_WITH,
  });
}

export async function getActiveCategories() {
  return db.select().from(categories).orderBy(categories.name);
}

export async function getActiveProductsByCategorySlug(categorySlug: string) {
  const category = await db.query.categories.findFirst({
    where: eq(categories.slug, categorySlug),
  });
  if (!category) return { category: null, products: [] };
  const productList = await db.query.products.findMany({
    where: and(eq(products.isActive, true), eq(products.categoryId, category.id)),
    ...ACTIVE_WITH,
    orderBy: [desc(products.createdAt)],
  });
  return { category, products: productList };
}
```
