test("addItem twice with same productId+size increments", () => {
  reset();
  const { addItem } = useCartStore.getState();

  assert.equal(s.items.length, 1);
  assert.equal(s.items[0].quantity, 2);

  reset();

  assert.equal(s.items.length, 2);

  incrementItem("p1::M");
  assert.equal(useCartStore.getState().items[0].quantity, 2);
  decrementItem("p1::M");
  assert.equal(useCartStore.getState().items[0].quantity, 1);

  assert.equal(useCartStore.getState().items.length, 0);

```typescript
import { create } from "zustand";
import { persist, createJSONStorage } from "zustand/middleware";

export type Size = "S" | "M" | "L";

export type CartItem = {
  key: string;                // `${productId}::${size}`
  productId: string;
  productSlug: string;
  name: string;
  image: string | null;       // relative /uploads/... path
  size: Size;
  variantId: string;
  unitPrice: string;          // Drizzle-shaped decimal string preserved verbatim
  quantity: number;
};

type AddItemInput = Omit<CartItem, "key" | "quantity">;

type CartState = {
  items: CartItem[];
  isDrawerOpen: boolean;
  addItem: (item: AddItemInput) => void;
  incrementItem: (key: string) => void;
  decrementItem: (key: string) => void;
  removeItem: (key: string) => void;
  setDrawerOpen: (open: boolean) => void;
  clear: () => void;
  getSubtotal: () => number;
  getItemCount: () => number;
};

const lineKey = (productId: string, size: Size) => `${productId}::${size}`;
const MAX_PER_LINE = 10; // per D2-20 soft cap

const isBrowser = typeof window !== "undefined";

export const useCartStore = create<CartState>()(
  persist(
    (set, get) => ({
      items: [],
      isDrawerOpen: false,

      addItem: (item) => {
        const key = lineKey(item.productId, item.size);
        const existing = get().items.find((i) => i.key === key);
        if (existing) {
          set({
            items: get().items.map((i) =>
              i.key === key ? { ...i, quantity: Math.min(i.quantity + 1, MAX_PER_LINE) } : i,
            ),
          });
          return;
        }
        set({ items: [...get().items, { ...item, key, quantity: 1 }] });
      },

      incrementItem: (key) =>
        set({
          items: get().items.map((i) =>
            i.key === key ? { ...i, quantity: Math.min(i.quantity + 1, MAX_PER_LINE) } : i,
          ),
        }),

      decrementItem: (key) => {
        const current = get().items.find((i) => i.key === key);
        if (!current) return;
        if (current.quantity <= 1) {
          set({ items: get().items.filter((i) => i.key !== key) });
          return;
        }
        set({
          items: get().items.map((i) =>
            i.key === key ? { ...i, quantity: i.quantity - 1 } : i,
          ),
        });
      },

      removeItem: (key) => set({ items: get().items.filter((i) => i.key !== key) }),

      setDrawerOpen: (open) => set({ isDrawerOpen: open }),

      clear: () => set({ items: [], isDrawerOpen: false }),

      getSubtotal: () =>
        get().items.reduce((sum, i) => {
          const price = parseFloat(i.unitPrice);
          if (!Number.isFinite(price)) return sum;
          return sum + price * i.quantity;
        }, 0),

      getItemCount: () => get().items.reduce((sum, i) => sum + i.quantity, 0),
    }),
    {
      name: "print-ninjaz-cart-v1", // D2-16 persistence key
      storage: createJSONStorage(() => (isBrowser ? localStorage : ({ getItem: () => null, setItem: () => {}, removeItem: () => {} } as Storage))),
      version: 1,
      partialize: (state) => ({ items: state.items }), // do NOT persist drawer open state
    },
  ),
);
```

The `isBrowser` / dummy storage guard makes the store safe to import in server component files (so a page that imports a component that imports the store doesn't crash SSR). The store doesn't render anything server-side.

| Boundary | Description |
|----------|-------------|
| Browser localStorage ↔ Zustand persist | Cart state is stored in an attacker-controllable medium on the user's device |
| Browser → `/cart`, `/checkout` links | User-driven navigation using data derived from localStorage |
| Client Add-to-bag → cart store | Product + variant data captured from DOM state into localStorage |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-04-01 | Tampering | localStorage `print-ninjaz-cart-v1` | mitigate | Cart is client-side UX convenience only. Authoritative price + variant resolution happens in Phase 3 on the server at checkout — server re-fetches the variant by ID, ignores client-supplied `unitPrice`, and rejects inactive variants. Phase 2 subtotal is purely display. |
| T-02-04-02 | Tampering | Quantity controls | mitigate | Line quantity clamped 0–10 client-side (MAX_PER_LINE). Phase 3 checkout will re-validate per-line quantity server-side before creating an order. |
| T-02-04-03 | Information Disclosure | localStorage cart contents | accept | Cart lines store product names, slugs, prices — all already public catalog data. No session tokens, user IDs, or PII are stored in the cart. |
| T-02-04-04 | Denial of Service | Zustand state bloat | mitigate | MAX_PER_LINE (10) + no line cap; but every line is one product+size, so lines are bounded by `product_count × 3 sizes`. Trivial memory footprint. Persist `partialize` excludes transient UI state (isDrawerOpen). |
| T-02-04-05 | Spoofing | Hydration mismatch (SSR count 0 vs client count N) | mitigate | CartButton and CartPage both use an `isMounted` guard (useEffect) so the SSR pass renders zero-count markup and the client swaps in the real count post-hydration. No React hydration warnings. |
| T-02-04-06 | Repudiation | Orphaned line items for deleted products | accept | If admin deletes a product the user has in their cart, the cart line still renders (product name/image from the snapshot) — clicking it 404s on the PDP. Phase 3 checkout validation will drop orphan lines with an explanatory toast before taking payment. Acceptable per D2 open-question resolution. |
