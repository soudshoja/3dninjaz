3. Create `src/actions/categories.ts` with handler-level admin auth checks:
```typescript
"use server";

import { auth } from "@/lib/auth";
import { headers } from "next/headers";
import { db } from "@/lib/db";
import { categories } from "@/lib/db/schema";
import { eq } from "drizzle-orm";
import { revalidatePath } from "next/cache";
import { categorySchema } from "@/lib/validators";

async function requireAdmin() {
  const session = await auth.api.getSession({ headers: await headers() });
  if (!session || session.user.role !== "admin") {
    throw new Error("Forbidden");
  }
  return session;
}

export async function createCategory(formData: FormData) {
  await requireAdmin();
  const parsed = categorySchema.safeParse({ name: formData.get("name") });
  if (!parsed.success) return { error: parsed.error.flatten().fieldErrors };
  
  const name = parsed.data.name;
  const slug = name.toLowerCase().replace(/\s+/g, "-").replace(/[^a-z0-9-]/g, "");
  
  try {
    await db.insert(categories).values({ name, slug });
  } catch (e: any) {
    if (e.message?.includes("unique")) return { error: { name: ["Category already exists"] } };
    throw e;
  }
  
  revalidatePath("/admin/categories");
  return { success: true };
}

export async function getCategories() {
  return db.select().from(categories).orderBy(categories.name);
}

export async function deleteCategory(id: string) {
  await requireAdmin();
  await db.delete(categories).where(eq(categories.id, id));
  revalidatePath("/admin/categories");
  return { success: true };
}
```

4. Create `src/actions/products.ts` with handler-level admin auth checks:
```typescript
"use server";

import { auth } from "@/lib/auth";
import { headers } from "next/headers";
import { db } from "@/lib/db";
import { products, productVariants } from "@/lib/db/schema";
import { eq } from "drizzle-orm";
import { revalidatePath } from "next/cache";
import { productSchema } from "@/lib/validators";

async function requireAdmin() {
  const session = await auth.api.getSession({ headers: await headers() });
  if (!session || session.user.role !== "admin") {
    throw new Error("Forbidden");
  }
  return session;
}

export async function createProduct(data: z.infer<typeof productSchema>) {
  await requireAdmin();
  const parsed = productSchema.safeParse(data);
  if (!parsed.success) return { error: parsed.error.flatten().fieldErrors };
  
  const { variants, ...productData } = parsed.data;
  const slug = productData.name.toLowerCase().replace(/\s+/g, "-").replace(/[^a-z0-9-]/g, "") + "-" + Date.now().toString(36);
  
  const [product] = await db.insert(products).values({
    ...productData,
    slug,
    categoryId: productData.categoryId || null,
    materialType: productData.materialType || null,
    estimatedProductionDays: productData.estimatedProductionDays || null,
  }).returning();
  
  // Insert variants
  if (variants.length > 0) {
    await db.insert(productVariants).values(
      variants.map(v => ({
        productId: product.id,
        size: v.size,
        price: v.price,
        widthCm: v.widthCm || null,
        heightCm: v.heightCm || null,
        depthCm: v.depthCm || null,
      }))
    );
  }
  
  revalidatePath("/admin/products");
  return { success: true, productId: product.id };
}

export async function updateProduct(id: string, data: z.infer<typeof productSchema>) {
  await requireAdmin();
  const parsed = productSchema.safeParse(data);
  if (!parsed.success) return { error: parsed.error.flatten().fieldErrors };
  
  const { variants, ...productData } = parsed.data;
  
  await db.update(products).set({
    ...productData,
    categoryId: productData.categoryId || null,
    materialType: productData.materialType || null,
    estimatedProductionDays: productData.estimatedProductionDays || null,
    updatedAt: new Date(),
  }).where(eq(products.id, id));
  
  // Replace variants: delete old, insert new
  await db.delete(productVariants).where(eq(productVariants.productId, id));
  if (variants.length > 0) {
    await db.insert(productVariants).values(
      variants.map(v => ({
        productId: id,
        size: v.size,
        price: v.price,
        widthCm: v.widthCm || null,
        heightCm: v.heightCm || null,
        depthCm: v.depthCm || null,
      }))
    );
  }
  
  revalidatePath("/admin/products");
  return { success: true };
}

export async function deleteProduct(id: string) {
  await requireAdmin();
  await db.delete(products).where(eq(products.id, id));
  // productVariants cascade-deleted via schema FK
  revalidatePath("/admin/products");
  return { success: true };
}

export async function toggleProductActive(id: string, isActive: boolean) {
  await requireAdmin();
  await db.update(products).set({ isActive, updatedAt: new Date() }).where(eq(products.id, id));
  revalidatePath("/admin/products");
  return { success: true };
}

export async function toggleProductFeatured(id: string, isFeatured: boolean) {
  await requireAdmin();
  await db.update(products).set({ isFeatured, updatedAt: new Date() }).where(eq(products.id, id));
  revalidatePath("/admin/products");
  return { success: true };
}

export async function getProduct(id: string) {
  return db.query.products.findFirst({
    where: eq(products.id, id),
    with: { variants: true, category: true },
  });
}

export async function getProducts() {
  return db.query.products.findMany({
    with: { variants: true, category: true },
    orderBy: (products, { desc }) => [desc(products.createdAt)],
  });
}
```

IMPORTANT: Every exported function that mutates data MUST call `await requireAdmin()` as its FIRST line. This is the CVE-2025-29927 mitigation. Read-only functions (getProduct, getProducts, getCategories) do not need admin check since product data is not sensitive.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Admin UI -> Server Actions | Form data and action parameters from admin browser session |
| Server Actions -> Database | Validated data written to cPanel MySQL |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Elevation of Privilege | Server actions | mitigate | Every mutation action calls requireAdmin() which verifies session.user.role === "admin" at handler level (CVE-2025-29927 mitigation). Layout redirect is convenience only, not security boundary. |
| T-03-02 | Tampering | Product data | mitigate | All input validated via Zod schemas (productSchema, categorySchema) before database writes. Max 5 images enforced at validation level per D-03. |
| T-03-03 | Tampering | Category slug | mitigate | Slug generated server-side from category name, not accepted from client input. Prevents slug injection. |
| T-03-04 | Denial of Service | Delete cascade | accept | Deleting a product cascades to variants (by FK design). This is intentional behavior. Admin-only action with confirmation dialog. |
| T-03-05 | Information Disclosure | Product data via getProducts | accept | Product data is not sensitive (will be public on storefront). Read-only actions do not require admin auth. |
</threat_model>
