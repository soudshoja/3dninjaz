1. Create `src/lib/storage.ts` — server-only filesystem helpers:
```typescript
import "server-only";
import { promises as fs } from "node:fs";
import path from "node:path";
import crypto from "node:crypto";

const UPLOADS_DIR = process.env.UPLOADS_DIR ?? "./public/uploads";
const PUBLIC_PREFIX = process.env.UPLOADS_PUBLIC_PREFIX ?? "/uploads";
const ALLOWED = new Set(["image/jpeg", "image/png", "image/webp"]);
const MAX_BYTES = 5 * 1024 * 1024; // 5MB

export async function writeUpload(bucket: string, file: File): Promise<string> {
  if (!ALLOWED.has(file.type)) throw new Error("Unsupported image type");
  if (file.size > MAX_BYTES) throw new Error("File too large (max 5MB)");

  const ext = file.type === "image/png" ? "png" : file.type === "image/webp" ? "webp" : "jpg";
  const name = `${crypto.randomUUID()}.${ext}`;
  const dir = path.join(process.cwd(), UPLOADS_DIR, "products", bucket);
  await fs.mkdir(dir, { recursive: true });

  const bytes = Buffer.from(await file.arrayBuffer());
  await fs.writeFile(path.join(dir, name), bytes);

  return `${PUBLIC_PREFIX}/products/${bucket}/${name}`;
}

export async function deleteUpload(publicUrl: string): Promise<void> {
  if (!publicUrl.startsWith(PUBLIC_PREFIX + "/")) return;
  const rel = publicUrl.slice(PUBLIC_PREFIX.length + 1);
  const abs = path.join(process.cwd(), UPLOADS_DIR, rel);
  // Guard against path traversal: resolved path must stay inside UPLOADS_DIR
  const root = path.resolve(path.join(process.cwd(), UPLOADS_DIR));
  if (!path.resolve(abs).startsWith(root)) return;
  try { await fs.unlink(abs); } catch {}
}

export function publicUrlFor(bucket: string, filename: string): string {
  return `${PUBLIC_PREFIX}/products/${bucket}/${filename}`;
}
```

2. Create `src/actions/uploads.ts` — server actions behind `writeUpload`. ALL actions MUST call `requireAdmin()` (reuse the helper pattern from Plan 03):
```typescript
"use server";
import { requireAdmin } from "@/lib/auth-helpers"; // or inline the same check
import { writeUpload, deleteUpload } from "@/lib/storage";

export async function uploadProductImage(formData: FormData): Promise<{ url?: string; error?: string }> {
  await requireAdmin();
  const bucket = String(formData.get("productId") ?? "new"); // use "new" for pre-create uploads
  const file = formData.get("file");
  if (!(file instanceof File)) return { error: "No file provided" };
  try {
    const url = await writeUpload(bucket, file);
    return { url };
  } catch (e: any) {
    return { error: e.message ?? "Upload failed" };
  }
}

export async function deleteProductImage(url: string): Promise<{ success: boolean }> {
  await requireAdmin();
  await deleteUpload(url);
  return { success: true };
}
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Tampering | Image upload | mitigate | Server action validates MIME type (jpg/png/webp only), size (<=5MB), and count (<=5). File extension is derived from MIME, not client-supplied name. Filename is a freshly generated UUID, never the uploaded name, to prevent path traversal or overwrite. `requireAdmin()` gate on every upload action. |
| T-04-02 | Tampering | Product form data | mitigate | All data validated server-side via Zod productSchema before database write. Client validation is convenience only. Images array validated as URL strings, max 5. |
| T-04-03 | Elevation of Privilege | Product create/edit | mitigate | createProduct and updateProduct both call requireAdmin() before any DB operation (CVE-2025-29927 mitigation from Plan 03). |
| T-04-04 | Denial of Service | Large image uploads | mitigate | Server enforces 5MB per file; admin-only route; cPanel disk quota caps total storage. Monitor via cPanel Disk Usage. |
| T-04-05 | Information Disclosure | Uploaded file URLs | accept | Upload paths are public and predictable by design (served by Next.js static handler). Only admin-uploaded product imagery lives here. No user-uploaded content. |
| T-04-06 | Tampering | Path traversal | mitigate | `deleteUpload` resolves the absolute path and rejects anything outside `UPLOADS_DIR`. Bucket name is either a UUID (product id) or the literal "new". |
</threat_model>
