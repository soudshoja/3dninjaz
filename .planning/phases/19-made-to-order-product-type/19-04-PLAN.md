    ```typescript
    export async function getConfiguratorData(productId: string) {
      await requireAdmin();
      const [product] = await db.select({
        id: products.id, name: products.name, slug: products.slug,
        productType: products.productType,
        maxUnitCount: products.maxUnitCount,
        priceTiersRaw: products.priceTiers,
        unitField: products.unitField,
      }).from(products).where(eq(products.id, productId)).limit(1);
      if (!product) throw new Error("Product not found");
      const rows = await db.select().from(productConfigFields)
        .where(eq(productConfigFields.productId, productId))
        .orderBy(asc(productConfigFields.position));
      const fields: ConfigField[] = rows.map((r) => ({
        id: r.id, productId: r.productId, position: r.position,
        fieldType: r.fieldType as FieldType,
        label: r.label,
        helpText: r.helpText,
        required: r.required,
        config: ensureConfigJson(r.fieldType as FieldType, r.configJson),
        createdAt: r.createdAt, updatedAt: r.updatedAt,
      }));
      return {
        product: {
          ...product,
          priceTiers: ensureTiers(product.priceTiersRaw),
        },
        fields,
      };
    }

    export async function addConfigField(productId, input) {
      await requireAdmin();
      const schema = pickSchemaByFieldType(input.fieldType);
      const parsed = schema.safeParse(input.config);
      if (!parsed.success) return { ok: false, error: parsed.error.issues[0]?.message ?? "invalid config" };
      const id = randomUUID();
      const [{ pos }] = await db.select({ pos: sql<number>`COALESCE(MAX(position),-1)+1` })
        .from(productConfigFields).where(eq(productConfigFields.productId, productId));
      await db.insert(productConfigFields).values({
        id, productId, position: pos,
        fieldType: input.fieldType,
        label: input.label,
        helpText: input.helpText ?? null,
        required: input.required,
        configJson: JSON.stringify(parsed.data),
      });
      revalidatePath(`/admin/products/${productId}/configurator`);
      // re-fetch the row for immediate caller use:
      const [row] = await db.select().from(productConfigFields).where(eq(productConfigFields.id, id)).limit(1);
      return { ok: true as const, field: hydrate(row) };
    }

    // updateConfigField — re-validate only when patch.config provided
    // deleteConfigField — db.delete().where(eq(productConfigFields.id, fieldId))
    // reorderConfigFields — db.transaction; SELECT all rows for product to verify orderedIds.length == count and Set equality; then UPDATE position per index.
    ```
