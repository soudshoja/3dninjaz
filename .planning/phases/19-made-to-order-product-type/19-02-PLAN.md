```typescript
function ensureImagesArray(raw: unknown): string[] {
  if (Array.isArray(raw)) return raw.filter((v): v is string => typeof v === "string");
  if (typeof raw === "string") {
    if (raw.trim() === "") return [];
    try {
      const parsed = JSON.parse(raw);
      if (Array.isArray(parsed)) {
        return parsed.filter((v): v is string => typeof v === "string");
      }
    } catch {
      return [];
    }
  }
  return [];
}
```

```typescript
// Per D-03 — Zod schemas + types per fieldType
export const TextFieldConfigSchema: z.ZodType<TextFieldConfig>;
export const NumberFieldConfigSchema: z.ZodType<NumberFieldConfig>;
export const ColourFieldConfigSchema: z.ZodType<ColourFieldConfig>;
export const SelectFieldConfigSchema: z.ZodType<SelectFieldConfig>;

export type TextFieldConfig    = { maxLength: number; allowedChars: string; uppercase: boolean; profanityCheck: boolean };
export type NumberFieldConfig  = { min: number; max: number; step: number };
export type ColourFieldConfig  = { allowedColorIds: string[] };
export type SelectFieldConfig  = { options: Array<{ label: string; value: string; priceAdd?: number }> };
export type FieldType = "text" | "number" | "colour" | "select";
export type AnyFieldConfig = TextFieldConfig | NumberFieldConfig | ColourFieldConfig | SelectFieldConfig;

// Returns parsed config object on success, throws on parse failure.
// Consumers that want fail-soft wrap in try/catch.
export function ensureConfigJson(fieldType: FieldType, raw: unknown): AnyFieldConfig;

// Returns price tiers map; falsy/invalid input returns {}.
export function ensureTiers(raw: unknown): Record<string, number>;

// Per D-05 — backwards-compat read for products.images.
// Old shape: ["url1", "url2"]. New shape: [{url, caption?, alt?, widths?, formats?}].
// Always returns the new shape; old strings auto-promoted to {url: s}.
export type ImageEntryV2 = { url: string; caption?: string; alt?: string };
export function ensureImagesV2(raw: unknown): ImageEntryV2[];

// Per D-11/D-12 — cart-line / order_item configurationData snapshot.
export type ConfigurationData = {
  values: Record<string, string>;     // fieldId -> value
  computedPrice: number;              // MYR
  computedSummary: string;            // "JACOB (5 letters) · Red base+chain · White letters"
};
export function ensureConfigurationData(raw: unknown): ConfigurationData | null;

// Per D-04 — tier lookup by length of the unitField value.
// Returns null when length is outside the table → caller disables Add-to-bag.
export function lookupTierPrice(
  tiers: Record<string, number>,
  unitFieldValue: string,
): number | null;
```
