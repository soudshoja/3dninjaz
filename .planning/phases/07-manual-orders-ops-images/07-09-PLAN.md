       ```typescript
       import { NextRequest, NextResponse } from "next/server";

       const ALLOWLIST_PREFIXES = [
         "/admin",              // admin can still log in
         "/api/paypal/webhook", // must keep capturing payments
         "/api/health",
         "/api/events/track",
         "/maintenance",        // avoid redirect loop
         "/_next",
         "/uploads",
         "/favicon",
       ];

       function isMaintenanceOn() {
         const v = (process.env.MAINTENANCE_MODE ?? "").toLowerCase();
         return v === "true" || v === "1" || v === "yes";
       }

       export function middleware(req: NextRequest) {
         if (!isMaintenanceOn()) return NextResponse.next();
         const { pathname } = req.nextUrl;
         if (ALLOWLIST_PREFIXES.some((p) => pathname === p || pathname.startsWith(p + "/") || pathname.startsWith(p))) {
           return NextResponse.next();
         }
         const url = req.nextUrl.clone();
         url.pathname = "/maintenance";
         return NextResponse.redirect(url);
       }

       export const config = {
         matcher: ["/((?!_next/static|_next/image).*)"],
       };
       ```

## Trust Boundaries
| Boundary | Description |
|----------|-------------|
| Any request -> middleware | Untrusted; middleware runs on every matched path |
| Thrown error -> error.tsx | Untrusted input (the error) must never reach the UI |

## STRIDE Threat Register
| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-09-error-page-leak | Information Disclosure | branded-500 | mitigate | Props strictly typed: only requestId + reset. `error` never passed to the rendered component. Verify via grep assertion for "error.message" and "error.stack" strings in the branded-500 source. |
| T-07-09-maintenance-bypass | Spoofing | middleware allowlist | accept | Admin-path allowlist is intentional (admin must log in to flip the flag back). Webhook allowlist is intentional (payments must keep capturing). No real bypass — allowlisted endpoints are safe by design. |
| T-07-09-DoS | Denial of Service | middleware matcher | accept | Runs on every request; simple string prefix check is O(n=8). Negligible cost. |
| T-07-09-request-id-spoof | Tampering | requestId generated client-side | accept | RequestId is 8-char random hex, client-generated. Purpose is LOG CORRELATION ONLY — not a security credential. Server logs its own copy. No trust placed in client value. |
