       ```typescript
       import "server-only";
       import sharp from "sharp";
       import { promises as fs } from "node:fs";
       import path from "node:path";

       const WIDTHS = [400, 800, 1600] as const;
       const MAX_INPUT_BYTES = 10 * 1024 * 1024;
       const MAX_PIXEL_AREA = 100_000_000;   // 100M pixels (10000x10000)

       export type ImageManifest = {
         widths: number[];
         variants: Array<{ width: number; webp: string; avif: string; jpg: string }>;
         original: string;   // filename of orig
       };

       function sniffMime(buf: Buffer): string | null {
         if (buf.length < 12) return null;
         if (buf.slice(0,3).toString("hex") === "ffd8ff") return "image/jpeg";
         if (buf.slice(0,8).toString("hex") === "89504e470d0a1a0a") return "image/png";
         if (buf.slice(0,4).toString("ascii") === "RIFF" && buf.slice(8,12).toString("ascii") === "WEBP") return "image/webp";
         if (buf.slice(0,4).toString("ascii") === "GIF8") return "image/gif";
         return null;
       }

       export async function compressUploadedImage(buf: Buffer, baseDir: string, opts: { skipBackup?: boolean } = {}): Promise<ImageManifest> {
         if (buf.length > MAX_INPUT_BYTES) throw new Error("Image exceeds 10MB cap");
         const mime = sniffMime(buf);
         if (!mime) throw new Error("Unrecognized image format");

         await fs.mkdir(baseDir, { recursive: true });
         const ext = mime === "image/jpeg" ? "jpg" : mime === "image/png" ? "png" : mime === "image/webp" ? "webp" : "gif";

         const meta = await sharp(buf, { failOn: "error" }).metadata();
         if ((meta.width ?? 0) * (meta.height ?? 0) > MAX_PIXEL_AREA) throw new Error("Image dimensions too large");
         // Animated GIF guard: pages * delay > 5000ms => reject
         if (meta.pages && meta.pages > 1 && meta.delay && meta.delay.reduce((a,b)=>a+b,0) > 5000) throw new Error("Animated GIF longer than 5s rejected");

         if (!opts.skipBackup) {
           await fs.writeFile(path.join(baseDir, `orig.${ext}`), buf);
         }

         const sourceWidth = meta.width ?? 0;
         const variants: ImageManifest["variants"] = [];
         for (const w of WIDTHS) {
           if (w > sourceWidth + 1) continue;   // skip upscale
           const pipeline = () => sharp(buf, { failOn: "error" }).rotate().resize({ width: w, withoutEnlargement: true });
           const webpPath = `${w}w.webp`, avifPath = `${w}w.avif`, jpgPath = `${w}w.jpg`;
           await pipeline().webp({ quality: 78 }).toFile(path.join(baseDir, webpPath));
           await pipeline().avif({ quality: 60 }).toFile(path.join(baseDir, avifPath));
           await pipeline().jpeg({ quality: 82, mozjpeg: true }).toFile(path.join(baseDir, jpgPath));
           variants.push({ width: w, webp: webpPath, avif: avifPath, jpg: jpgPath });
         }
         // Always include at least one variant — if source < 400px, downscale to source-width
         if (variants.length === 0) {
           const w = sourceWidth || 400;
           const pipeline = () => sharp(buf, { failOn: "error" }).rotate().resize({ width: w, withoutEnlargement: true });
           await pipeline().webp({ quality: 78 }).toFile(path.join(baseDir, `${w}w.webp`));
           await pipeline().avif({ quality: 60 }).toFile(path.join(baseDir, `${w}w.avif`));
           await pipeline().jpeg({ quality: 82, mozjpeg: true }).toFile(path.join(baseDir, `${w}w.jpg`));
           variants.push({ width: w, webp: `${w}w.webp`, avif: `${w}w.avif`, jpg: `${w}w.jpg` });
         }

         const manifest: ImageManifest = {
           widths: variants.map((v)=>v.width),
           variants,
           original: `orig.${ext}`,
         };
         await fs.writeFile(path.join(baseDir, "manifest.json"), JSON.stringify(manifest));
         return manifest;
       }
       ```

    2. **`src/lib/image-manifest.ts`** — server-only:
       ```typescript
       import "server-only";
       import { promises as fs } from "node:fs";
       import path from "node:path";

       export type PictureSource = { type: "image/avif"|"image/webp"|"image/jpeg"; srcSet: string; src?: string };
       export type PictureData = { sources: PictureSource[]; fallbackSrc: string; alt?: string };

       function isLegacyUrl(url: string): boolean {
         return /\.(jpe?g|png|webp|gif)$/i.test(url);
       }

       export async function pickImage(baseUrl: string, opts: { uploadsDir?: string; publicPrefix?: string } = {}): Promise<PictureData> {
         const publicPrefix = opts.publicPrefix ?? "/uploads";
         const uploadsRoot = opts.uploadsDir ?? path.join(process.cwd(), "public", "uploads");
         if (isLegacyUrl(baseUrl)) {
           // No manifest — emit single jpg/png/webp source
           return { sources: [], fallbackSrc: baseUrl };
         }
         const rel = baseUrl.startsWith(publicPrefix) ? baseUrl.slice(publicPrefix.length+1) : baseUrl.replace(/^\//,"");
         const dir = path.join(uploadsRoot, rel);
         try {
           const m = JSON.parse(await fs.readFile(path.join(dir, "manifest.json"), "utf8"));
           const widths = m.widths as number[];
           const variants = m.variants as Array<{width:number; webp:string; avif:string; jpg:string}>;
           const baseUrlSlash = baseUrl.endsWith("/") ? baseUrl : baseUrl + "/";
           const srcSet = (key: "webp"|"avif"|"jpg") =>
             variants.map((v) => `${baseUrlSlash}${v[key]} ${v.width}w`).join(", ");
           const sources: PictureSource[] = [
             { type: "image/avif", srcSet: srcSet("avif") },
             { type: "image/webp", srcSet: srcSet("webp") },
             { type: "image/jpeg", srcSet: srcSet("jpg") },
           ];
           const fallbackVariant = variants[Math.max(0, variants.length-1)];
           return { sources, fallbackSrc: `${baseUrlSlash}${fallbackVariant.jpg}` };
         } catch {
           return { sources: [], fallbackSrc: baseUrl };   // safe fallback
         }
       }
       ```

       ```tsx
       import "server-only";
       import { pickImage } from "@/lib/image-manifest";

       export async function ResponsiveProductImage({
         imageUrl, alt, sizes = "(max-width: 768px) 100vw, 33vw", className, eager,
       }: { imageUrl: string; alt: string; sizes?: string; className?: string; eager?: boolean }) {
         const data = await pickImage(imageUrl);
         if (data.sources.length === 0) {
           return <img src={data.fallbackSrc} alt={alt} className={className} loading={eager ? "eager" : "lazy"} />;
         }
         return (
           <picture>
             {data.sources.map((s) => (
               <source key={s.type} type={s.type} srcSet={s.srcSet} sizes={sizes} />
             ))}
             <img src={data.fallbackSrc} alt={alt} className={className} loading={eager ? "eager" : "lazy"} />
           </picture>
         );
       }
       ```
