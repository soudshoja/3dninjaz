    2. **Create `scripts/phase7-migrate.cjs`** mirroring `scripts/phase6-migrate.cjs` precedent (per Phase 6 06-01 SUMMARY: drizzle-kit push hangs on remote MariaDB). Use mysql2/promise + dotenv:
       ```javascript
       require("dotenv").config({ path: ".env.local" });
       const mysql = require("mysql2/promise");

       async function tableExists(conn, name) { /* SHOW TABLES LIKE ? */ }
       async function columnExists(conn, table, col) { /* SHOW COLUMNS FROM ? LIKE ? */ }

       async function main() {
         const url = new URL(process.env.DATABASE_URL);
         const conn = await mysql.createConnection({
           host: url.hostname,
           port: Number(url.port || 3306),
           user: decodeURIComponent(url.username),
           password: decodeURIComponent(url.password),
           database: url.pathname.replace(/^\//, ""),
           multipleStatements: false,
         });

         // Step 1 — additive columns on orders (each guarded by columnExists)
         const orderCols = [
           ["source_type", "ENUM('web','manual') NOT NULL DEFAULT 'web'"],
           ["custom_item_name", "VARCHAR(200) NULL"],
           ["custom_item_description", "TEXT NULL"],
           ["custom_images", "LONGTEXT NULL"],   // MariaDB JSON stored as LONGTEXT (CLAUDE.md quirk)
           ["refunded_amount", "DECIMAL(10,2) NOT NULL DEFAULT '0.00'"],
           ["paypal_fee", "DECIMAL(10,2) NULL"],
           ["paypal_net", "DECIMAL(10,2) NULL"],
           ["seller_protection", "VARCHAR(32) NULL"],
           ["paypal_settle_date", "TIMESTAMP NULL"],
         ];
         for (const [col, type] of orderCols) {
           if (await columnExists(conn, "orders", col)) { console.log(`skip orders.${col}`); continue; }
           await conn.query(`ALTER TABLE orders ADD COLUMN ${col} ${type}`);
           console.log(`applied orders.${col}`);
         }

         // Step 2 — CREATE TABLE IF NOT EXISTS for payment_links, dispute_cache, recon_runs
         // CHARSET=latin1 to match existing user/orders FK constraints (Phase 6 precedent).
         await conn.query(`
           CREATE TABLE IF NOT EXISTS payment_links (
             id VARCHAR(36) PRIMARY KEY,
             order_id VARCHAR(36) NOT NULL,
             token VARCHAR(64) NOT NULL UNIQUE,
             expires_at TIMESTAMP NOT NULL,
             used_at TIMESTAMP NULL,
             created_by VARCHAR(36) NOT NULL,
             created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
             INDEX payment_links_order_idx (order_id),
             CONSTRAINT payment_links_order_fk FOREIGN KEY (order_id) REFERENCES orders(id) ON DELETE CASCADE,
             CONSTRAINT payment_links_user_fk FOREIGN KEY (created_by) REFERENCES user(id)
           ) ENGINE=InnoDB DEFAULT CHARSET=latin1
         `);

         await conn.query(`
           CREATE TABLE IF NOT EXISTS dispute_cache (
             id VARCHAR(36) PRIMARY KEY,
             dispute_id VARCHAR(64) NOT NULL UNIQUE,
             order_id VARCHAR(36) NULL,
             status VARCHAR(32) NOT NULL,
             reason VARCHAR(64),
             amount DECIMAL(10,2),
             currency VARCHAR(3),
             create_date TIMESTAMP NOT NULL,
             update_date TIMESTAMP NOT NULL,
             last_synced_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
             raw_json LONGTEXT NULL,
             INDEX dispute_cache_status_idx (status),
             INDEX dispute_cache_order_idx (order_id),
             CONSTRAINT dispute_cache_order_fk FOREIGN KEY (order_id) REFERENCES orders(id) ON DELETE SET NULL
           ) ENGINE=InnoDB DEFAULT CHARSET=latin1
         `);

         await conn.query(`
           CREATE TABLE IF NOT EXISTS recon_runs (
             id VARCHAR(36) PRIMARY KEY,
             run_date DATE NOT NULL UNIQUE,
             ran_at TIMESTAMP NOT NULL,
             total_paypal_txns INT NOT NULL,
             total_local_txns INT NOT NULL,
             drift_count INT NOT NULL DEFAULT 0,
             drift_json LONGTEXT NULL,
             status VARCHAR(16) NOT NULL,
             error_message TEXT NULL
           ) ENGINE=InnoDB DEFAULT CHARSET=latin1
         `);

         await conn.end();
         console.log("phase7-migrate complete");
       }
       main().catch((e) => { console.error(e); process.exit(1); });
       ```
