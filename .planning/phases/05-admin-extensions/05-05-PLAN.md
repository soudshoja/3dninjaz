       import "server-only";
       import { parse } from "csv-parse/sync";
       import { readFile } from "node:fs/promises";

       const MAX_BYTES = 5 * 1024 * 1024; // 5 MB
       const MAX_ROWS = 1000;

       export type CsvRow = Record<string, string>;

       export async function parseCsvStream(filePath: string): Promise<CsvRow[]> {
         const stat = await (await import("node:fs/promises")).stat(filePath);
         if (stat.size > MAX_BYTES) throw new Error(`CSV too large: ${stat.size} bytes (max ${MAX_BYTES})`);
         const buf = await readFile(filePath);
         const rows = parse(buf, {
           columns: true,
           skip_empty_lines: true,
           trim: true,
           relax_quotes: true,
           relax_column_count: false, // strict column count
         }) as CsvRow[];
         if (rows.length > MAX_ROWS) throw new Error(`Too many rows: ${rows.length} (max ${MAX_ROWS})`);
         return rows;
       }

       export function normalizeCsvRow(raw: CsvRow): NormalizedResult {
         // Lowercase keys for consistency
         const row: Record<string, string> = {};
         for (const [k, v] of Object.entries(raw)) row[k.toLowerCase().replace(/\s+/g, "_")] = v;

         // Extract image URLs
         const imageKeys = Object.keys(row).filter(k => /^image_url_\d+$/.test(k));
         const images: string[] = [];
         const errors: string[] = [];
         for (const k of imageKeys) {
           const v = row[k]?.trim();
           if (!v) continue;
           if (v.startsWith("http://") || v.startsWith("https://")) {
             errors.push(`${k}: external URLs not allowed — upload images via product form`);
             continue;
           }
           if (!v.startsWith("/uploads/products/")) {
             errors.push(`${k}: must be a /uploads/products/... path`);
             continue;
           }
           images.push(v);
         }

         return { row, images, errors };
       }

       type NormalizedResult = { row: Record<string,string>; images: string[]; errors: string[] };
       ```

       "use server";
       import { db } from "@/lib/db";
       import { products, productVariants, categories } from "@/lib/db/schema";
       import { eq, inArray } from "drizzle-orm";
       import { randomUUID } from "crypto";
       import { revalidatePath } from "next/cache";
       import { requireAdmin } from "@/lib/auth-helpers";
       import { bulkImportRowSchema } from "@/lib/validators";
       import { parseCsvStream, normalizeCsvRow, type CsvRow } from "@/lib/csv";

       export type PreviewResult = {
         validRows: Array<{ rowIndex: number; data: ReturnType<typeof normalize> }>;
         invalidRows: Array<{ rowIndex: number; errors: string[]; raw: CsvRow }>;
         summary: { total: number; valid: number; invalid: number };
       };

       export async function previewCsv(fileName: string): Promise<{ ok: true; result: PreviewResult } | { ok: false; error: string }> {
         await requireAdmin();
         // Resolve fileName inside public/uploads/imports (guard against path traversal)
         const safeName = fileName.replace(/[^a-zA-Z0-9-.]/g, "");
         if (!safeName.endsWith(".csv")) return { ok: false, error: "Not a CSV file" };
         const path = `${process.cwd()}/public/uploads/imports/${safeName}`;
         let rows: CsvRow[];
         try {
           rows = await parseCsvStream(path);
         } catch (e) {
           return { ok: false, error: e instanceof Error ? e.message : "Parse failed" };
         }

         // Load existing slugs for dedup
         const existingSlugs = new Set((await db.select({ slug: products.slug }).from(products)).map(r => r.slug));
         const csvSlugs = new Set<string>();
         const categoriesByName = new Map((await db.select().from(categories)).map(c => [c.name.toLowerCase(), c.id]));

         const validRows: PreviewResult["validRows"] = [];
         const invalidRows: PreviewResult["invalidRows"] = [];

         rows.forEach((raw, i) => {
           const { row, images, errors: rowErrors } = normalizeCsvRow(raw);
           const errors: string[] = [...rowErrors];

           // Zod validate
           const zodParse = bulkImportRowSchema.safeParse(row);
           if (!zodParse.success) errors.push(...zodParse.error.issues.map(iss => `${iss.path.join(".")}: ${iss.message}`));

           // Derive slug if not supplied
           const slug = (row.slug || slugify(row.name || "")).trim();
           if (!slug) errors.push("slug: cannot derive from name");
           if (existingSlugs.has(slug)) errors.push(`slug: "${slug}" already exists in DB`);
           if (csvSlugs.has(slug)) errors.push(`slug: "${slug}" appears more than once in this CSV`);
           csvSlugs.add(slug);

           // Resolve categoryId from category_name
           let categoryId: string | null = null;
           if (row.category_name) {
             const found = categoriesByName.get(row.category_name.toLowerCase());
             if (!found) errors.push(`category_name: "${row.category_name}" not found — create category first`);
             else categoryId = found;
           }

           if (errors.length > 0) {
             invalidRows.push({ rowIndex: i + 2 /* +2 for header + 1-index */, errors, raw });
           } else {
             validRows.push({ rowIndex: i + 2, data: { ...zodParse.data, slug, categoryId, images } });
           }
         });

         return {
           ok: true,
           result: {
             validRows, invalidRows,
             summary: { total: rows.length, valid: validRows.length, invalid: invalidRows.length },
           },
         };
       }

       export async function commitCsvImport(fileName: string) {
         await requireAdmin();
         const prev = await previewCsv(fileName);
         if (!prev.ok) return { ok: false, error: prev.error };

         const results: { successes: string[]; failures: Array<{row: number; error: string}> } = {
           successes: [], failures: [],
         };

         await db.transaction(async (tx) => {
           for (const v of prev.result.validRows) {
             const productId = randomUUID();
             try {
               await tx.insert(products).values({
                 id: productId,
                 name: v.data.name,
                 slug: v.data.slug,
                 description: v.data.description,
                 images: v.data.images as string[],
                 materialType: v.data.material_type ?? null,
                 estimatedProductionDays: v.data.estimated_production_days ?? null,
                 categoryId: v.data.categoryId ?? null,
                 isActive: true,
                 isFeatured: false,
               });
               const variants: Array<{id:string;productId:string;size:"S"|"M"|"L";price:string}> = [];
               if (v.data.price_s) variants.push({ id: randomUUID(), productId, size: "S", price: v.data.price_s });
               if (v.data.price_m) variants.push({ id: randomUUID(), productId, size: "M", price: v.data.price_m });
               if (v.data.price_l) variants.push({ id: randomUUID(), productId, size: "L", price: v.data.price_l });
               if (variants.length > 0) await tx.insert(productVariants).values(variants);
               results.successes.push(v.data.slug);
             } catch (e) {
               results.failures.push({ row: v.rowIndex, error: e instanceof Error ? e.message : "Insert failed" });
               throw e; // Rollback everything
             }
           }
         }).catch(() => { /* transaction already rolled back */ });

         // Delete the temp CSV file after commit (success or failure)
         try { await (await import("node:fs/promises")).unlink(`${process.cwd()}/public/uploads/imports/${fileName}`); } catch {}

         revalidatePath("/admin/products");
         return { ok: true, results };
       }

       function slugify(s: string): string {
         return s.toLowerCase().replace(/[^a-z0-9]+/g, "-").replace(/^-|-$/g, "").slice(0, 220);
       }
       ```

       ```typescript
       import { NextRequest } from "next/server";
       import { writeFile, mkdir } from "node:fs/promises";
       import { randomUUID } from "crypto";
       import { requireAdmin } from "@/lib/auth-helpers";

       const MAX_BYTES = 5 * 1024 * 1024;
       const ALLOWED_MIMES = ["text/csv", "application/vnd.ms-excel", "application/csv"];

       export async function POST(req: NextRequest) {
         try {
           await requireAdmin(); // T-05-05-EoP
         } catch {
           return new Response(JSON.stringify({ error: "Forbidden" }), { status: 403 });
         }

         const form = await req.formData();
         const file = form.get("file");
         if (!(file instanceof File)) return new Response(JSON.stringify({ error: "No file" }), { status: 400 });

         if (file.size > MAX_BYTES) return new Response(JSON.stringify({ error: "Too large" }), { status: 413 });
         if (!ALLOWED_MIMES.includes(file.type) && !file.name.endsWith(".csv")) {
           return new Response(JSON.stringify({ error: "CSV files only" }), { status: 400 });
         }

         const fileName = `${randomUUID()}.csv`;
         const dir = `${process.cwd()}/public/uploads/imports`;
         await mkdir(dir, { recursive: true });
         const buf = Buffer.from(await file.arrayBuffer());
         await writeFile(`${dir}/${fileName}`, buf);

         return new Response(JSON.stringify({ fileName }), { status: 200 });
       }
       ```
