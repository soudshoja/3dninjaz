       ```typescript
       "use server";
       import { db } from "@/lib/db";
       import { user, orders } from "@/lib/db/schema";
       import { eq, ne, sql, desc, count } from "drizzle-orm";
       import { revalidatePath } from "next/cache";
       import { requireAdmin } from "@/lib/auth-helpers";
       import { userSuspendSchema } from "@/lib/validators";

       export type AdminUserRow = {
         id: string; name: string; email: string;
         role: string; banned: boolean; banReason: string | null; banExpires: Date | null;
         createdAt: Date; orderCount: number;
       };

       export async function listAdminUsers(): Promise<AdminUserRow[]> {
         await requireAdmin();
         // 2 SELECTs: users with role != "admin" + order counts
         const users = await db.select().from(user).where(ne(user.role, "admin")).orderBy(desc(user.createdAt));
         if (users.length === 0) return [];
         const ids = users.map(u => u.id);
         const orderCounts = await db.select({ userId: orders.userId, c: count() })
           .from(orders)
           .where(sql`${orders.userId} IN (${sql.join(ids.map(id => sql`${id}`), sql`, `)})`)
           .groupBy(orders.userId);
         const countMap = new Map(orderCounts.map(r => [r.userId, Number(r.c)]));
         return users.map(u => ({
           id: u.id, name: u.name, email: u.email, role: u.role,
           banned: !!u.banned, banReason: u.banReason ?? null, banExpires: u.banExpires ?? null,
           createdAt: u.createdAt, orderCount: countMap.get(u.id) ?? 0,
         }));
       }

       export async function suspendUser(formData: FormData): Promise<{ ok: true } | { ok: false; error: string }> {
         const session = await requireAdmin();
         const parsed = userSuspendSchema.safeParse({
           userId: formData.get("userId"),
           suspend: formData.get("suspend") === "true",
           reason: formData.get("reason"),
         });
         if (!parsed.success) return { ok: false, error: parsed.error.issues[0].message };
         const { userId, suspend, reason } = parsed.data;

         // T-05-02-EoP: prevent self-suspend + admin-on-admin
         if (userId === session.user.id) return { ok: false, error: "Cannot suspend yourself" };
         const [target] = await db.select({ role: user.role }).from(user).where(eq(user.id, userId)).limit(1);
         if (!target) return { ok: false, error: "User not found" };
         if (target.role === "admin") return { ok: false, error: "Cannot suspend an admin user" };

         await db.update(user).set({
           banned: suspend,
           banReason: suspend ? (reason ?? null) : null,
           banExpires: null, // v1: permanent suspension until explicit unsuspend; future add date picker
         }).where(eq(user.id, userId));

         revalidatePath("/admin/users");
         return { ok: true };
       }

       export async function unsuspendUser(userId: string): Promise<{ ok: true } | { ok: false; error: string }> {
         await requireAdmin();
         const [target] = await db.select({ id: user.id }).from(user).where(eq(user.id, userId)).limit(1);
         if (!target) return { ok: false, error: "User not found" };
         await db.update(user).set({ banned: false, banReason: null, banExpires: null }).where(eq(user.id, userId));
         revalidatePath("/admin/users");
         return { ok: true };
       }
       ```

       ```typescript
       export type AnalyticsRange = "7d" | "30d" | "90d";
       export function parseRange(val: string | null): AnalyticsRange {
         return val === "7d" || val === "90d" ? val : "30d";
       }
       export function rangeToDays(range: AnalyticsRange): number {

         d.setUTCDate(d.getUTCDate() - rangeToDays(range) + 1);

       }
       ```

    4. Create `src/actions/admin-analytics.ts`:

       ```typescript
       "use server";
       import { db } from "@/lib/db";
       import { orders, orderItems, session } from "@/lib/db/schema";
       import { and, gte, inArray, sql, desc } from "drizzle-orm";
       import { requireAdmin } from "@/lib/auth-helpers";
       import { parseRange, rangeStartDate, iterateDays, type AnalyticsRange } from "@/lib/analytics";

       const REVENUE_STATUSES = ["paid", "processing", "shipped", "delivered"] as const;

       export async function getAnalytics(range: AnalyticsRange) {
         await requireAdmin();
         const start = rangeStartDate(range);

         // 1. Revenue total + order count
         const [totals] = await db.select({
           revenue: sql<string>`COALESCE(SUM(${orders.totalAmount}), 0)`,
           orderCount: sql<number>`COUNT(*)`,
         }).from(orders)
         .where(and(
           inArray(orders.status, REVENUE_STATUSES),
           gte(orders.createdAt, start),
         ));

         // 2. Revenue by day (for chart)
         const revenueByDay = await db.select({
           day: sql<string>`DATE(${orders.createdAt})`,
           revenue: sql<string>`COALESCE(SUM(${orders.totalAmount}), 0)`,
         }).from(orders)
         .where(and(
           inArray(orders.status, REVENUE_STATUSES),
           gte(orders.createdAt, start),
         ))
         .groupBy(sql`DATE(${orders.createdAt})`);

         // 3. Fill missing days with zero (so chart shows continuous line)
         const dayMap = new Map(revenueByDay.map(r => [r.day, r.revenue]));
         const chartData = iterateDays(start, new Date()).map(day => ({
           day, revenue: parseFloat(dayMap.get(day) ?? "0"),
         }));

         // 4. Top products — join orders -> order_items, sum lineTotal by productId
         const topProducts = await db.select({
           productId: orderItems.productId,
           productName: orderItems.productName,
           totalRevenue: sql<string>`COALESCE(SUM(${orderItems.lineTotal}), 0)`,
           totalQuantity: sql<number>`COALESCE(SUM(${orderItems.quantity}), 0)`,
         }).from(orderItems)
         .innerJoin(orders, sql`${orders.id} = ${orderItems.orderId}`)
         .where(and(
           inArray(orders.status, REVENUE_STATUSES),
           gte(orders.createdAt, start),
         ))
         .groupBy(orderItems.productId, orderItems.productName)
         .orderBy(sql`SUM(${orderItems.lineTotal}) DESC`)
         .limit(5);

         // 5. Funnel — approximations for v1:
         //    - visits: distinct session count in range (from Better Auth session table)
         //    - addToBag: events table or pageViews — use /api/events/track writes (below)
         //    - checkoutStarted: orders WHERE status IN all (including pending) — i.e. count of create-order calls
         //    - paid: orders with status IN revenueStatuses
         const visits = await db.select({ c: sql<number>`COUNT(DISTINCT ${session.userId})` })
           .from(session)
           .where(gte(session.createdAt, start));
         const [checkoutStarted] = await db.select({ c: sql<number>`COUNT(*)` })
           .from(orders)
           .where(gte(orders.createdAt, start));
         const paid = totals.orderCount;
         // addToBag approximation: read page_views or a separate events table; if no events
         // table yet, return 0 with a "tracking not yet installed" flag.

         return {
           range,
           revenue: parseFloat(totals.revenue),
           orderCount: Number(totals.orderCount),
           chartData,
           topProducts,
           funnel: {
             visits: Number(visits[0]?.c ?? 0),
             addToBag: 0, // TODO: populated once events table rows exist
             checkoutStarted: Number(checkoutStarted.c),
             paid: Number(paid),
           },
         };
       }
       ```

       Note: The `addToBag` counter is intentionally 0 until events table fills — surface an "instrumentation pending" note in the UI if the value is 0 and /api/events/track has never been called.
