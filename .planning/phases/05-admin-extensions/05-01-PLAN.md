       export const couponSchema = z.object({
         code: z.string().regex(/^[A-Z0-9_-]{3,32}$/, "Code: 3-32 chars, A-Z, 0-9, _, -"),
         type: couponTypeEnum,
         amount: z.string().regex(/^\d+(\.\d{1,2})?$/, "Amount: decimal with up to 2 places"),
         minSpend: z.string().regex(/^\d+(\.\d{1,2})?$/).optional().nullable(),

         usageCap: z.coerce.number().int().positive().optional().nullable(),
         active: z.boolean().default(true),
       }).superRefine((val, ctx) => {
         if (val.type === "percentage" && parseFloat(val.amount) > 100) { ctx.addIssue({ code: "custom", path: ["amount"], message: "Percentage cannot exceed 100" }); }
