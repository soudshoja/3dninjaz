       ```typescript
       import "server-only";
       import DOMPurify from "isomorphic-dompurify";

       const ALLOWED_TAGS = ["a","b","br","div","em","h1","h2","h3","h4","hr","i","img","li","ol","p","span","strong","table","tbody","td","tfoot","th","thead","tr","u","ul"];
       const ALLOWED_ATTR = ["href","src","alt","title","target","rel","style","width","height","align","cellpadding","cellspacing","border","bgcolor","colspan","rowspan"];

       export function sanitiseEmailHtml(html: string): string {
         return DOMPurify.sanitize(html, {
           ALLOWED_TAGS, ALLOWED_ATTR,
           FORBID_ATTR: ["onerror","onload","onclick","onmouseover","onfocus","onblur"],
           ALLOW_DATA_ATTR: false,
         });
       }

       export function escapeHtml(v: unknown): string {
         const s = String(v);
         return s.replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;").replace(/"/g,"&quot;").replace(/'/g,"&#39;");
       }
       ```

       ```typescript
       import "server-only";
       import { db } from "@/lib/db";
       import { emailTemplates } from "@/lib/db/schema";
       import { eq } from "drizzle-orm";
       import { sanitiseEmailHtml, escapeHtml } from "@/lib/email/sanitize";

       export type TemplateKey = "order_confirmation" | "password_reset";

       export const availableVariables: Record<TemplateKey, string[]> = {
         order_confirmation: ["customer_name", "order_number", "order_total", "order_link", "items_table"],
         password_reset: ["customer_name", "reset_link"],
       };

       const DEFAULTS: Record<TemplateKey, { subject: string; html: string }> = {
         order_confirmation: {
           subject: "Your 3D Ninjaz order {{order_number}} is confirmed",
           html: `<!DOCTYPE html><html><body><h1>Thanks, {{customer_name}}!</h1><p>Your order {{order_number}} is confirmed. Total: {{order_total}}.</p>{{items_table}}<p><a href="{{order_link}}">View your order</a></p></body></html>`,
         },
         password_reset: {
           subject: "Reset your 3D Ninjaz password",
           html: `<!DOCTYPE html><html><body><h1>Hi {{customer_name}},</h1><p>Click below to reset your password:</p><p><a href="{{reset_link}}">Reset password</a></p></body></html>`,
         },
       };

       async function getOrSeed(key: TemplateKey) {
         const [row] = await db.select().from(emailTemplates).where(eq(emailTemplates.key, key)).limit(1);
         if (row) return row;
         const seed = { key, subject: DEFAULTS[key].subject, html: DEFAULTS[key].html, variables: availableVariables[key] };
         await db.insert(emailTemplates).values(seed);
         return seed;
       }

       export async function renderTemplate(
         key: TemplateKey,
         variables: Record<string, unknown>,
       ): Promise<{ subject: string; html: string; text: string }> {
         const tpl = await getOrSeed(key);
         // Sanitise stored html again as defense-in-depth (already sanitised on save)
         const safeHtml = sanitiseEmailHtml(tpl.html);
         // Substitute variables with HTML-escaped values
         const substitute = (src: string) =>
           src.replace(/\{\{(\w+)\}\}/g, (_, name) => escapeHtml(variables[name] ?? ""));
         const html = substitute(safeHtml);
         const subject = tpl.subject.replace(/\{\{(\w+)\}\}/g, (_, name) => String(variables[name] ?? ""));
         const text = htmlToText(html);
         return { subject, html, text };
       }

       /** Minimal HTML → text fallback for email clients that prefer plain text. */
       function htmlToText(html: string): string {
         return html
           .replace(/<style[^>]*>[\s\S]*?<\/style>/gi, "")
           .replace(/<script[^>]*>[\s\S]*?<\/script>/gi, "")
           .replace(/<\/?[^>]+>/g, "")
           .replace(/&nbsp;/g, " ")
           .replace(/&amp;/g, "&")
           .replace(/\s+/g, " ")
           .trim();
       }
       ```

       ```typescript
       const HTML_VARS = new Set(["items_table"]); // caller guarantees these are sanitised
       const substitute = (src: string) =>
         src.replace(/\{\{(\w+)\}\}/g, (_, name) => {
           const val = variables[name] ?? "";
           return HTML_VARS.has(name) ? sanitiseEmailHtml(String(val)) : escapeHtml(val);
         });
       ```

       ```typescript
       "use server";
       import { db } from "@/lib/db";
       import { emailTemplates } from "@/lib/db/schema";
       import { eq } from "drizzle-orm";
       import { revalidatePath } from "next/cache";
       import { requireAdmin } from "@/lib/auth-helpers";
       import { emailTemplateSchema } from "@/lib/validators";
       import { sanitiseEmailHtml } from "@/lib/email/sanitize";
       import { availableVariables, type TemplateKey } from "@/lib/email/templates";

       export async function listEmailTemplates() {
         await requireAdmin();
         return await db.select().from(emailTemplates);
       }

       export async function getEmailTemplate(key: string) {
         await requireAdmin();
         const [row] = await db.select().from(emailTemplates).where(eq(emailTemplates.key, key)).limit(1);
         return row ?? null;
       }

       export async function updateEmailTemplate(formData: FormData) {
         await requireAdmin();
         const parsed = emailTemplateSchema.safeParse({
           key: formData.get("key"),
           subject: formData.get("subject"),
           html: formData.get("html"),
         });
         if (!parsed.success) return { ok: false, error: parsed.error.issues[0].message };
         const { key, subject, html } = parsed.data;

         // T-05-06-HTML: sanitise before save
         const safeHtml = sanitiseEmailHtml(html);

         await db.update(emailTemplates).set({
           subject, html: safeHtml,
           variables: availableVariables[key as TemplateKey] ?? [],
         }).where(eq(emailTemplates.key, key));

         revalidatePath("/admin/email-templates");
         revalidatePath(`/admin/email-templates/${key}/edit`);
         return { ok: true };
       }
       ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| admin session → email template editor | requireAdmin() guard |
| stored html → email render | Sanitised on save; re-sanitised on render |
| variable value (customer name, order number) → html substitution | HTML-escaped by default |
| preview iframe → admin browser | srcDoc + sandbox="" (no scripts) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-06-EoP | Elevation of Privilege | admin-email-templates actions | mitigate | requireAdmin() FIRST await in every export |
| T-05-06-HTML | Tampering / XSS | admin edits template HTML | mitigate | DOMPurify with strict allowlist on save AND on render; variable vals HTML-escaped; preview iframe sandbox="" (no script execution) |
| T-05-06-XSS-variable | Tampering | variable substitution | mitigate | Default: HTML-escape all variable values; only explicitly-marked HTML_VARS (e.g., items_table) skip escaping and are sanitised |
| T-05-06-DoS | DoS | html upload size | mitigate | 100KB cap in Zod |
| T-05-06-clickjacking | Spoofing | preview iframe | mitigate | srcDoc (not URL); sandbox="" blocks all navigation and scripts |
| T-05-06-CSRF | Tampering | save template | mitigate | Server action invocation has built-in CSRF protection via Next.js |
| T-05-06-stale-template | Information disclosure | cached stored template | accept | No cache; always read from DB. Low perf cost (1 query per email, templates are 2 rows) |
</threat_model>
