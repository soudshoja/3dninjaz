    2. **`src/app/(store)/orders/[id]/invoice.pdf/route.ts`** — GET handler:
       ```ts
       import { NextResponse } from "next/server";
       import { requireUser } from "@/lib/auth-helpers";
       import { getMyOrder } from "@/actions/orders";
       import { formatOrderNumber } from "@/lib/orders";
       import ReactPDF from "@react-pdf/renderer";
       import { InvoiceDocument } from "@/lib/pdf/invoice";
       import { BUSINESS } from "@/lib/business-info";

       // In-process rate limit — v1 single-instance (Phase 3 03-03 precedent)
       const INVOICE_COOLDOWN_MS = 60 * 60 * 1000; // 1h
       const INVOICE_MAX_PER_WINDOW = 10;
       const invoiceLog = new Map<string, number[]>(); // userId -> [timestamps...]

       function allowInvoice(userId: string): boolean {
         const now = Date.now();
         const arr = (invoiceLog.get(userId) ?? []).filter(t => now - t < INVOICE_COOLDOWN_MS);
         if (arr.length >= INVOICE_MAX_PER_WINDOW) {
           invoiceLog.set(userId, arr); // prune
           return false;
         }
         arr.push(now);
         invoiceLog.set(userId, arr);
         return true;
       }

       export async function GET(req: Request, { params }: { params: Promise<{ id: string }> }) {
         const session = await requireUser(); // T-06-06-auth
         const { id } = await params;

         if (!allowInvoice(session.user.id)) {
           return new NextResponse("Too many invoice downloads. Try again later.", { status: 429 });
         }

         const order = await getMyOrder(id);
         if (!order) {
           return new NextResponse("Not found", { status: 404 }); // T-06-06-IDOR — same response as truly missing
         }

         // Prefer DB-backed settings if Phase 5 05-04 is merged; else fall back to static BUSINESS.
         let business = {
           businessName: BUSINESS.legalName,
           contactEmail: BUSINESS.contactEmail,
           whatsappDisplay: BUSINESS.whatsappNumberDisplay,
           pdpaLine: "This is a digital invoice; no signature required. Records retained for 7 years per Malaysian PDPA 2010.",
         };
         try {
           const mod = await import("@/actions/store-settings");
           if (typeof (mod as { getStoreSettings?: () => Promise<unknown> }).getStoreSettings === "function") {
             const s = await (mod as { getStoreSettings: () => Promise<{ businessName: string; contactEmail: string; whatsappNumberDisplay: string }> }).getStoreSettings();
             if (s) {
               business = { businessName: s.businessName, contactEmail: s.contactEmail, whatsappDisplay: s.whatsappNumberDisplay, pdpaLine: business.pdpaLine };
             }
           }
         } catch {
           /* Phase 5 05-04 not merged — use BUSINESS fallback */
         }

         const stream = await ReactPDF.renderToStream(<InvoiceDocument order={order as never} business={business} />);
         const chunks: Buffer[] = [];
         for await (const chunk of stream as unknown as AsyncIterable<Buffer>) chunks.push(chunk);
         const buffer = Buffer.concat(chunks);

         return new NextResponse(buffer, {
           status: 200,
           headers: {
             "Content-Type": "application/pdf",
             "Content-Disposition": `inline; filename="invoice-${formatOrderNumber(order.id)}.pdf"`,
             "Cache-Control": "private, no-store",
           },
         });
       }
       ```

    1. **`src/actions/order-requests.ts`** — customer-side:
       ```ts
       "use server";
       import { requireUser } from "@/lib/auth-helpers";
       import { db } from "@/lib/db";
       import { orderRequests, orders } from "@/lib/db/schema";
       import { orderRequestSchema } from "@/lib/validators";
       import { and, eq, desc } from "drizzle-orm";
       import { randomUUID } from "node:crypto";
       import { revalidatePath } from "next/cache";

       const RETURN_WINDOW_MS = 14 * 24 * 60 * 60 * 1000; // 14 days (Assumption per CUST-07)

       export async function listMyOrderRequests(orderId: string) {
         const session = await requireUser();
         // Ownership check via orders.userId — a user cannot list requests on another user's order.
         const [order] = await db.select({ userId: orders.userId }).from(orders).where(eq(orders.id, orderId)).limit(1);
         if (!order || order.userId !== session.user.id) return [];
         return db
           .select()
           .from(orderRequests)
           .where(eq(orderRequests.orderId, orderId))
           .orderBy(desc(orderRequests.createdAt));
       }

       export async function submitOrderRequest(input: unknown) {
         const session = await requireUser();
         const parsed = orderRequestSchema.safeParse(input);
         if (!parsed.success) return { ok: false as const, error: parsed.error.issues[0].message };

         // Ownership gate via orders table (same SELECT pattern as getMyOrder)
         const [order] = await db
           .select({ id: orders.id, userId: orders.userId, status: orders.status, updatedAt: orders.updatedAt })
           .from(orders)
           .where(eq(orders.id, parsed.data.orderId))
           .limit(1);
         if (!order || order.userId !== session.user.id) return { ok: false as const, error: "Order not found." }; // same response (T-06-06-enumeration)

         // Eligibility rules
         if (parsed.data.type === "cancel") {
           if (!(order.status === "pending" || order.status === "paid")) {
             return { ok: false as const, error: "Cancel is only available before the order ships." };
           }
         } else {
           // return
           if (order.status !== "delivered") {
             return { ok: false as const, error: "Return is only available after the order is delivered." };
           }
           const ageMs = Date.now() - new Date(order.updatedAt).getTime();
           if (ageMs > RETURN_WINDOW_MS) {
             return { ok: false as const, error: "Return window closed. Contact support via WhatsApp." };
           }
         }

         // One-pending-per-order rule (Assumption 7)
         const [existingPending] = await db
           .select({ id: orderRequests.id })
           .from(orderRequests)
           .where(and(eq(orderRequests.orderId, parsed.data.orderId), eq(orderRequests.status, "pending")))
           .limit(1);
         if (existingPending) {
           return { ok: false as const, error: "You already have a pending request on this order." };
         }

         await db.insert(orderRequests).values({
           id: randomUUID(),
           orderId: parsed.data.orderId,
           userId: session.user.id,
           type: parsed.data.type,
           reason: parsed.data.reason,
           status: "pending",
         });

         revalidatePath(`/orders/${parsed.data.orderId}`);
         revalidatePath(`/admin/orders/${parsed.data.orderId}`);
         return { ok: true as const, message: parsed.data.type === "cancel" ? "Cancel request submitted. We'll review it shortly." : "Return request submitted. We'll review it shortly." };
       }
       ```

       ```ts
       import { requireAdmin } from "@/lib/auth-helpers";
       import { db } from "@/lib/db";
       import { orderRequests, orders } from "@/lib/db/schema";

       export async function listOrderRequestsForOrder(orderId: string) {
         await requireAdmin();
         return db.select().from(orderRequests).where(eq(orderRequests.orderId, orderId)).orderBy(desc(orderRequests.createdAt));
       }

       export async function approveOrderRequest(requestId: string, adminNotes?: string) {
         await requireAdmin();
         return db.transaction(async (tx) => {
           const [req] = await tx.select().from(orderRequests).where(eq(orderRequests.id, requestId)).limit(1);
           if (!req) return { ok: false as const, error: "Request not found." };
           if (req.status !== "pending") return { ok: false as const, error: "Request is not pending." };

           if (req.type === "cancel") {
             // Flip order.status -> cancelled. assertValidTransition will throw if the order moved past the cancellable point.
             const [order] = await tx.select({ status: orders.status }).from(orders).where(eq(orders.id, req.orderId)).limit(1);
             if (!order) return { ok: false as const, error: "Order not found." };
             try {
               assertValidTransition(order.status as never, "cancelled");
             } catch {
               return { ok: false as const, error: "Order has already shipped — cancel no longer valid." };
             }
             await tx.update(orders).set({ status: "cancelled" }).where(eq(orders.id, req.orderId));
           }
           // For return: order.status stays 'delivered' per Assumption 6; only the request transitions.

           await tx.update(orderRequests).set({
             status: "approved",
             adminNotes: adminNotes ?? null,
             resolvedAt: new Date(),
           }).where(eq(orderRequests.id, requestId));

           return { ok: true as const };
         }).then((res) => {
           revalidatePath(`/admin/orders`);
           // revalidate all orders detail pages — we don't know orderId without fetching; cheap enough
           return res;
         });
       }

       export async function rejectOrderRequest(requestId: string, adminNotes?: string) {
         await requireAdmin();
         const [req] = await db.select().from(orderRequests).where(eq(orderRequests.id, requestId)).limit(1);
         if (!req) return { ok: false as const, error: "Request not found." };
         if (req.status !== "pending") return { ok: false as const, error: "Request is not pending." };
         await db.update(orderRequests).set({
           status: "rejected",
           adminNotes: adminNotes ?? null,
           resolvedAt: new Date(),
         }).where(eq(orderRequests.id, requestId));
         revalidatePath(`/admin/orders/${req.orderId}`);
         revalidatePath(`/orders/${req.orderId}`);
         return { ok: true as const };
       }
       ```
