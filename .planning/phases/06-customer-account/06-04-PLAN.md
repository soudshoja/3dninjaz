       ```ts
       "use server";
       import { requireUser, getSessionUser } from "@/lib/auth-helpers";
       import { db } from "@/lib/db";
       import { wishlists, products, productVariants } from "@/lib/db/schema";
       import { wishlistAddSchema } from "@/lib/validators";
       import { eq, and, inArray, asc, desc } from "drizzle-orm";
       import { randomUUID } from "node:crypto";
       import { revalidatePath } from "next/cache";

       export async function isWishlisted(productId: string): Promise<boolean> {
         // Called from server components during PDP render. NOT from client.
         const user = await getSessionUser();
         if (!user) return false;
         const [row] = await db
           .select({ id: wishlists.id })
           .from(wishlists)
           .where(and(eq(wishlists.userId, user.id), eq(wishlists.productId, productId)))
           .limit(1);
         return !!row;
       }

       export async function toggleWishlist(input: unknown): Promise<{ ok: true; state: "added" | "removed" } | { ok: false; error: string }> {
         const session = await requireUser();
         const parsed = wishlistAddSchema.safeParse(input);
         if (!parsed.success) return { ok: false, error: "Invalid product id." };

         // Existence check + toggle in two queries. UNIQUE(user_id, product_id) makes
         // double-insert races land as constraint violations, which we catch.
         const [existing] = await db
           .select({ id: wishlists.id })
           .from(wishlists)
           .where(and(eq(wishlists.userId, session.user.id), eq(wishlists.productId, parsed.data.productId)))
           .limit(1);

         if (existing) {
           await db.delete(wishlists).where(eq(wishlists.id, existing.id));
           revalidatePath("/account/wishlist");
           return { ok: true, state: "removed" };
         }

         try {
           await db.insert(wishlists).values({
             id: randomUUID(),
             userId: session.user.id,
             productId: parsed.data.productId,
           });
         } catch (err) {
           // UNIQUE constraint race — treat as already-added idempotency
           if (err && typeof err === "object" && "code" in err && (err as { code?: string }).code === "ER_DUP_ENTRY") {
             return { ok: true, state: "added" };
           }
           console.error("[toggleWishlist] insert failed", err);
           return { ok: false, error: "Could not update wishlist." };
         }

         revalidatePath("/account/wishlist");
         return { ok: true, state: "added" };
       }

       export async function removeFromWishlist(productId: string) {
         const session = await requireUser();
         await db
           .delete(wishlists)
           .where(and(eq(wishlists.userId, session.user.id), eq(wishlists.productId, productId)));
         revalidatePath("/account/wishlist");
         return { ok: true as const };
       }

       export async function listMyWishlist() {
         const session = await requireUser();
         // Manual hydration — MariaDB 10.11 rejects LATERAL joins (see 03-01 SUMMARY).
         const rows = await db
           .select({
             id: wishlists.id,
             productId: wishlists.productId,
             createdAt: wishlists.createdAt,
           })
           .from(wishlists)
           .where(eq(wishlists.userId, session.user.id))
           .orderBy(desc(wishlists.createdAt));

         if (rows.length === 0) return [];

         const productIds = rows.map(r => r.productId);
         const productRows = await db
           .select()
           .from(products)
           .where(and(inArray(products.id, productIds), eq(products.isActive, true)));

         // Hydrate variants in a single IN query, then group.
         const variants = productIds.length
           ? await db
               .select()
               .from(productVariants)
               .where(inArray(productVariants.productId, productIds))
               .orderBy(asc(productVariants.size))
           : [];

         // Normalize images (MariaDB quirks from Phase 3 02 SUMMARY — string vs array)
         const ensureImages = (v: unknown): string[] => {
           if (Array.isArray(v)) return v.filter((x): x is string => typeof x === "string");
           if (typeof v === "string") { try { const p = JSON.parse(v); return Array.isArray(p) ? p.filter((x): x is string => typeof x === "string") : []; } catch { return []; } }
           return [];
         };

         const productById = new Map(productRows.map(p => [p.id, { ...p, images: ensureImages(p.images) }]));
         const variantsByProduct = new Map<string, typeof variants>();
         for (const v of variants) {
           const arr = variantsByProduct.get(v.productId) ?? [];
           arr.push(v);
           variantsByProduct.set(v.productId, arr);
         }

         // Final shape — includes wishlist.createdAt so UI can show "Added 3 days ago" later.
         return rows
           .map(r => {
             const product = productById.get(r.productId);
             if (!product) return null; // product deleted or inactive — silently hide
             const pvs = variantsByProduct.get(r.productId) ?? [];
             return { wishlistId: r.id, wishlistedAt: r.createdAt, product, variants: pvs };
           })
           .filter((x): x is NonNullable<typeof x> => x !== null);
       }
       ```

    1. **`src/components/store/wishlist-button.tsx`** — client component with optimistic UI:
       ```tsx
       "use client";
       import { Heart } from "lucide-react";
       import { useState, useTransition } from "react";
       import { useRouter, usePathname } from "next/navigation";
       import { toggleWishlist } from "@/actions/wishlist";
       import { authClient } from "@/lib/auth-client";
       import { BRAND } from "@/lib/brand";

       type Variant = "overlay" | "pill" | "inline";

       export function WishlistButton({
         productId,
         initialState,
         variant = "overlay",
         className = "",
       }: {
         productId: string;
         initialState: boolean;
         variant?: Variant;
         className?: string;
       }) {
         const { data: session } = authClient.useSession();
         const router = useRouter();
         const pathname = usePathname();
         const [on, setOn] = useState(initialState);
         const [pending, startTransition] = useTransition();

         const handleClick = (e: React.MouseEvent) => {
           e.preventDefault();
           e.stopPropagation(); // CRITICAL on ProductCard — don't trigger the outer Link

           if (!session?.user) {
             router.push(`/login?next=${encodeURIComponent(pathname ?? "/")}`);
             return;
           }

           // Optimistic flip
           const next = !on;
           setOn(next);
           startTransition(async () => {
             const res = await toggleWishlist({ productId });
             if (!res.ok) {
               setOn(!next); // rollback
               console.error("[WishlistButton] toggle failed", res.error);
             } else if ((res.state === "added") !== next || (res.state === "removed") !== !next) {
               // Reconcile with server truth
               setOn(res.state === "added");
             }
           });
         };

         // Size rules per variant
         const size = variant === "overlay" ? "h-10 w-10" : variant === "pill" ? "h-11 px-4" : "h-12 w-12";
         const shape = variant === "pill" ? "rounded-full" : "rounded-full";
         // min-h enforcement — D-04 tap target rule
         const tap = variant === "overlay" ? "min-h-[44px] min-w-[44px]" : "min-h-[48px] min-w-[48px]";

         return (
           <button
             type="button"
             onClick={handleClick}
             disabled={pending}
             aria-pressed={on}
             aria-label={on ? "Remove from wishlist" : "Add to wishlist"}
             className={`inline-flex items-center justify-center gap-2 ${size} ${shape} ${tap} ${className}`}
             style={{
               backgroundColor: on ? `${BRAND.purple}25` : "rgba(255,255,255,0.9)",
               color: on ? BRAND.purple : BRAND.ink,
               border: `2px solid ${on ? BRAND.purple : BRAND.ink}`,
             }}
           >
             <Heart className="h-5 w-5" fill={on ? BRAND.purple : "transparent"} />
             {variant === "pill" ? <span className="text-sm font-bold">{on ? "Saved" : "Save"}</span> : null}
           </button>
         );
       }
       ```
