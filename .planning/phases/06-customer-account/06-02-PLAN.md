    1. **`src/actions/account.ts`** — `"use server"` at top. Three server actions, each starts with `await requireUser()`:
       ```ts
       "use server";
       import { requireUser } from "@/lib/auth-helpers";
       import { auth } from "@/lib/auth";
       import { db } from "@/lib/db";
       import { user } from "@/lib/db/schema";
       import { profileUpdateSchema, changeEmailSchema, changePasswordSchema } from "@/lib/validators";
       import { eq } from "drizzle-orm";
       import { headers } from "next/headers";
       import { revalidatePath } from "next/cache";

       export async function updateProfile(input: { name: string }) {
         const session = await requireUser();
         const parsed = profileUpdateSchema.safeParse(input);
         if (!parsed.success) return { ok: false as const, error: parsed.error.issues[0].message };
         await db.update(user).set({ name: parsed.data.name }).where(eq(user.id, session.user.id));
         revalidatePath("/account");
         return { ok: true as const };
       }

       export async function changeEmail(input: { newEmail: string; currentPassword: string }) {
         await requireUser(); // gate first (T-06-02-auth)
         const parsed = changeEmailSchema.safeParse(input);
         if (!parsed.success) return { ok: false as const, error: parsed.error.issues[0].message };
         // Better Auth native — triggers verification email to newEmail.
         // It re-auths currentPassword internally. We pass headers() for session discovery.
         try {
           await auth.api.changeEmail({
             body: { newEmail: parsed.data.newEmail, callbackURL: "/account/security?verified=1" },
             headers: await headers(),
           });
           return { ok: true as const, message: `Verification email sent to ${parsed.data.newEmail}. Click the link there to complete the change.` };
         } catch (err) {
           console.error("[changeEmail] Better Auth error", err);
           return { ok: false as const, error: "Could not start email change. Check your current password and try again." };
         }
       }

       export async function changePassword(input: { currentPassword: string; newPassword: string }) {
         await requireUser();
         const parsed = changePasswordSchema.safeParse(input);
         if (!parsed.success) return { ok: false as const, error: parsed.error.issues[0].message };
         try {
           await auth.api.changePassword({
             body: {
               currentPassword: parsed.data.currentPassword,
               newPassword: parsed.data.newPassword,
               revokeOtherSessions: false, // keep current session alive
             },
             headers: await headers(),
           });
           return { ok: true as const, message: "Password updated." };
         } catch (err) {
           console.error("[changePassword] Better Auth error", err);
           return { ok: false as const, error: "Could not change password. Check your current password and try again." };
         }
       }
       ```

       Threat mitigations:
       - T-06-02-auth: requireUser() is the FIRST await in every action — CVE-2025-29927 guard
       - T-06-02-credential-stuffing: Better Auth's changeEmail/changePassword both require currentPassword — no session-only bypass
       - T-06-02-PII-log: console.error logs only the error object; NEVER the password or newEmail

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Unauth HTTP -> /account/* | Layout redirect + requireUser() in actions; double gate |
| Client form -> updateProfile / changeEmail / changePassword | Server actions validate payload with Zod + call Better Auth APIs with headers() for session discovery |
| Better Auth -> SMTP (changeEmail verification) | Fire-and-forget pattern; SMTP outage is logged, not blocking the 200 response |
| Session cookie -> user.deletedAt | requireUser() checks deletedAt defensively even if Better Auth ban propagation is delayed (T-06-01-closure from 06-01) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-02-auth | Elevation of privilege | /account/* and all actions | mitigate | Layout redirects unauth; requireUser() is FIRST await in every action (CVE-2025-29927) |
| T-06-02-credential-stuffing | Spoofing | changeEmail, changePassword | mitigate | Better Auth re-auths currentPassword server-side; no session-only bypass |
| T-06-02-enumeration | Information disclosure | changeEmail error messages | mitigate | Generic error copy ("Could not start email change") — never leaks whether the new email is already in use or the password failed. If Better Auth returns a distinguishing error, strip it in the server action. |
