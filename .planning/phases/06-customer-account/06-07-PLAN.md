    1. **Harden `src/lib/auth-helpers.ts`** — 06-01 added `requireUser()` with a defensive `deletedAt` check on `session.user`. However, Better Auth may not surface custom columns on session.user by default. Add an explicit DB reload when `session.user.deletedAt` is undefined (missing, not null) — defense in depth against propagation lag and schema-level "user is logged in but row is soft-deleted" race:

       ```ts
       import "server-only";
       import { headers } from "next/headers";
       import { auth } from "@/lib/auth";
       import { db } from "@/lib/db";
       import { user as userTable } from "@/lib/db/schema";
       import { eq } from "drizzle-orm";

       export async function requireAdmin() {
         const session = await auth.api.getSession({ headers: await headers() });
         if (!session || session.user.role !== "admin") throw new Error("Forbidden");
         return session;
       }

       export async function getSessionUser() {
         const session = await auth.api.getSession({ headers: await headers() });
         return session?.user ?? null;
       }

       export async function requireUser() {
         const session = await auth.api.getSession({ headers: await headers() });
         if (!session) throw new Error("Unauthorized");

         // Hot-path check — Better Auth may expose deletedAt on session.user if configured.
         const surface = session.user as { deletedAt?: Date | string | null };
         if (surface.deletedAt) throw new Error("Unauthorized");

         // Cold-path — when session payload does not include deletedAt (typical for Better Auth
         // without an additionalFields declaration), reload the row once per request. T-06-07-lag.
         if (surface.deletedAt === undefined) {
           const [row] = await db
             .select({ deletedAt: userTable.deletedAt, banned: userTable.banned })
             .from(userTable)
             .where(eq(userTable.id, session.user.id))
             .limit(1);
           if (!row || row.deletedAt || row.banned) throw new Error("Unauthorized");
         }

         return session;
       }
       ```

       Threat mitigations:
       - T-06-07-lag: even if Better Auth hasn't yet propagated banUser across all session caches, the DB reload on each requireUser() call catches a session cookie that belongs to a closed account
       - T-06-07-perf: reload only fires when session.user.deletedAt is undefined; zero extra query for sessions that do surface the flag; at most 1 extra SELECT per request otherwise

    2. **`src/actions/account-close.ts`** — the destructive action:
       ```ts
       "use server";
       import { requireUser } from "@/lib/auth-helpers";
       import { auth } from "@/lib/auth";
       import { db } from "@/lib/db";
       import { user as userTable, addresses, wishlists, session as sessionTable } from "@/lib/db/schema";
       import { accountCloseSchema } from "@/lib/validators";
       import { eq } from "drizzle-orm";
       import { headers } from "next/headers";
       import { redirect } from "next/navigation";

       export async function closeMyAccount(input: unknown) {
         const s = await requireUser(); // T-06-07-auth

         const parsed = accountCloseSchema.safeParse(input);
         if (!parsed.success) return { ok: false as const, error: "Type DELETE to confirm." };

         const userId = s.user.id;
         const anonymizedEmail = `deleted-${userId}@3dninjaz.local`;

         // Atomic anonymize + customer-data wipe. Orders / order_requests / reviews preserved (PDPA D-06).
         // NOTE: user row is NOT deleted — UNIQUE email constraint + NO-cascade FKs from orders,
         // order_requests, reviews require the row to linger (with anonymized email).
         await db.transaction(async (tx) => {
           // Anonymize identity
           await tx.update(userTable).set({
             email: anonymizedEmail,
             emailVerified: false,
             name: "Former customer",
             image: null,
             banned: true,
             banReason: "Account closed by user (CUST-08)",
             deletedAt: new Date(),
           }).where(eq(userTable.id, userId));

           // Wipe customer-convenience data (no audit value)
           await tx.delete(addresses).where(eq(addresses.userId, userId));
           await tx.delete(wishlists).where(eq(wishlists.userId, userId));

           // Hard-kill all sessions for this user so the cookie is dead everywhere.
           // Better Auth's session table is keyed by userId (Phase 1 schema).
           await tx.delete(sessionTable).where(eq(sessionTable.userId, userId));
         });

         // Best-effort Better Auth admin plugin banUser — in case Better Auth has additional
         // internal state (in-memory session cache) beyond the DB session table. Try/catch so
         // failure doesn't undo our DB work; requireUser()'s deletedAt check is the backstop.
         try {
           if (auth.api.banUser) {
             await auth.api.banUser({ body: { userId, banReason: "Account closed by user" }, headers: await headers() });
           }
         } catch (err) {
           console.error("[closeMyAccount] banUser best-effort failed (DB state already closed)", err);
         }

         // Redirect to homepage with a closure flag — we're bypassing return because cookies are gone.
         redirect("/?closed=1");
       }
       ```

       Threat mitigations:
       - T-06-07-auth: requireUser() first await
       - T-06-07-consent: accountCloseSchema.literal("DELETE") blocks accidental/JS-coerced submits
       - T-06-07-atomicity: db.transaction wraps anonymize + cascade + session delete in one atomic unit
       - T-06-07-session-kill: delete from session table is the guaranteed kill; banUser is belt-and-braces for Better Auth's internal caches
       - T-06-07-PDPA: orders / order_requests / reviews preserved (customerEmail snapshot on orders already protects buyer contact for 7y)
       - T-06-07-PII-log: console.error logs only the error object default string; no PII
       - T-06-07-race: if requireUser() throws between form submit and this action firing (e.g. user was just banned by admin), the action never runs; idempotency preserved

    Threat mitigations embedded:
    - T-06-07-re-registration: anonymized email frees the original; user can re-register fresh (Assumption 7). Old orders stay anchored to the closed-and-anonymized row; new account is separate.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Closed-account session cookie -> requireUser() | Hot-path deletedAt surface check + cold-path DB reload — session cannot linger |
| DELETE literal gate -> closeMyAccount | Zod literal rejects any other value; button disabled client-side; server re-validates |
| Transaction -> anonymize + cascade + session-kill | Single atomic unit; no half-closed state possible |
| Better Auth banUser -> best-effort | Wrapped in try/catch; DB state already closed regardless of ban propagation |
| orders + order_requests + reviews FKs -> user | NO cascade; rows survive with anonymized user row anchoring PDPA audit |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-07-auth | Elevation of privilege | closeMyAccount | mitigate | requireUser() FIRST await (CVE-2025-29927) |
| T-06-07-consent | Tampering | accidental closure | mitigate | accountCloseSchema.literal("DELETE") + client button disabled until typed match + explicit copy |
| T-06-07-atomicity | Integrity | partial closure (anonymized but wishlist still visible) | mitigate | db.transaction wraps user update + addresses delete + wishlists delete + session delete |
| T-06-07-lag | Elevation of privilege | stale session after closure | mitigate | requireUser() cold-reload on missing session.user.deletedAt; session table delete + banUser belt-and-braces |
| T-06-07-PDPA-over-delete | Compliance | deleting orders violates 7y retention | mitigate | Orders NOT touched by closure; user row anonymized but kept so FK integrity survives |
| T-06-07-PDPA-under-delete | Compliance | leaking PII after user asks for closure | mitigate | email -> anonymized string, name -> "Former customer", image -> null; reviews attribute anonymized via 06-05 |
| T-06-07-re-registration | Tampering | closed user registers with stolen prior-email | accept | Re-registration with original email is allowed by design (Assumption 7); new account is logically disjoint from old; no order history leak because orders are FK-scoped by userId |
| T-06-07-social-engineering | Spoofing | user requests account recovery after closure | accept | Closure is immediate and irreversible by design; user can re-register. Support should not attempt to "reopen" closed accounts. Copy on /account/close sets expectations. |
| T-06-07-XSS | Tampering | closure banner on homepage | mitigate | React auto-escape; banner text is static |
| T-06-07-CSRF | Tampering | closure triggered by cross-site form | accept | Next.js server actions are same-origin; same-site cookie + POST-only is sufficient. Additional countermeasure (DELETE literal input) makes cross-site CSRF impractical even if origin check somehow bypassed. |
| T-06-07-PII-log | Information disclosure | logs leak email / name during closure | mitigate | console.error uses only error object default string; NEVER log email, name, or userId body |
| T-06-07-anonymize-collision | Integrity | two users with same UUID-derived anonymized email | accept | anonymized email format `deleted-<userId>@3dninjaz.local` uses the UUID PK — collision probability is negligible |
</threat_model>
