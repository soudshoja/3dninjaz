Next.js 15 file-based robots and sitemap conventions:

```ts
// src/app/robots.ts
import type { MetadataRoute } from "next";
import { SITE } from "@/lib/site-metadata";

export default function robots(): MetadataRoute.Robots {
  return {
    rules: [
      { userAgent: "*", allow: ["/", "/shop", "/products/", "/about", "/contact", "/privacy", "/terms"],
        disallow: ["/admin", "/api", "/orders", "/bag", "/checkout", "/login", "/register", "/forgot-password", "/reset-password"] },
    ],
    sitemap: `${SITE.url}/sitemap.xml`,
    host: SITE.url,
  };
}

// src/app/sitemap.ts
import type { MetadataRoute } from "next";
import { SITE } from "@/lib/site-metadata";
import { getActiveProducts } from "@/lib/catalog";

export default async function sitemap(): Promise<MetadataRoute.Sitemap> {
  const products = await getActiveProducts();
  const staticRoutes = ["", "/shop", "/about", "/contact", "/privacy", "/terms"].map((path) => ({
    url: `${SITE.url}${path}`,
    lastModified: new Date(),
    changeFrequency: "weekly" as const,
    priority: path === "" ? 1.0 : 0.7,
  }));
  const productRoutes = products.map((p) => ({
    url: `${SITE.url}/products/${p.slug}`,
    lastModified: p.updatedAt ?? new Date(),
    changeFrequency: "weekly" as const,
    priority: 0.8,
  }));
  return [...staticRoutes, ...productRoutes];
}
```

**`src/app/robots.ts`**

```ts
import type { MetadataRoute } from "next";
import { SITE } from "@/lib/site-metadata";

export default function robots(): MetadataRoute.Robots {
  return {
    rules: [
      {
        userAgent: "*",
        allow: ["/"],
        disallow: [
          "/admin",
          "/admin/",
          "/api/",
          "/orders",
          "/orders/",
          "/bag",
          "/checkout",
          "/login",
          "/register",
          "/forgot-password",
          "/reset-password",
        ],
      },
    ],
    sitemap: `${SITE.url}/sitemap.xml`,
    host: SITE.url,
  };
}
```

Rationale for disallow list:
- `/admin/*` — admin interface, never indexed
- `/api/*` — server endpoints, never indexed
- `/orders/*` — per-user, requires auth, personal data
- `/bag`, `/checkout` — transient shopping state, no SEO value, potential session leakage
- `/login`, `/register`, `/forgot-password`, `/reset-password` — auth forms, indexing them confuses search intent

Allow `/` catches homepage and all other public pages (Next.js generates the rules as flat text — `/shop`, `/products/...`, `/about`, etc. all fall under the default allow).

**`src/app/sitemap.ts`**

```ts
import type { MetadataRoute } from "next";
import { SITE } from "@/lib/site-metadata";
import { getActiveProducts } from "@/lib/catalog";

export default async function sitemap(): Promise<MetadataRoute.Sitemap> {
  let products: { slug: string; updatedAt?: Date }[] = [];
  try {
    products = await getActiveProducts();
  } catch (err) {
    // If DB is unreachable at build time, emit static routes only.
    // The sitemap will refresh on next successful request (dynamic route).
    console.warn("[sitemap] failed to load products:", err);
  }

  const now = new Date();
  const staticRoutes: MetadataRoute.Sitemap = [
    { url: `${SITE.url}/`, lastModified: now, changeFrequency: "weekly", priority: 1.0 },
    { url: `${SITE.url}/shop`, lastModified: now, changeFrequency: "daily", priority: 0.9 },
    { url: `${SITE.url}/about`, lastModified: now, changeFrequency: "monthly", priority: 0.6 },
    { url: `${SITE.url}/contact`, lastModified: now, changeFrequency: "monthly", priority: 0.6 },
    { url: `${SITE.url}/privacy`, lastModified: now, changeFrequency: "yearly", priority: 0.4 },
    { url: `${SITE.url}/terms`, lastModified: now, changeFrequency: "yearly", priority: 0.4 },
  ];

  const productRoutes: MetadataRoute.Sitemap = products.map((p) => ({
    url: `${SITE.url}/products/${p.slug}`,
    lastModified: p.updatedAt ?? now,
    changeFrequency: "weekly" as const,
    priority: 0.8,
  }));

  return [...staticRoutes, ...productRoutes];
}
```

**CRITICAL:** This MUST use `getActiveProducts` (products where isActive = true). Never emit inactive or draft product slugs to the sitemap — those are admin-only. Catalog helper from Phase 2 Plan 02-01 enforces this filter.

  <name>Task 2: Write .htaccess for HTTPS + HSTS + security headers (and proxy if option A)</name>
  <files>.htaccess</files>
  <action>
Write `.htaccess` at the project root for commit (will be deployed to `public_html/` on cPanel during the deploy task).

```apache
# Force HTTPS (redirects HTTP to HTTPS with 301)
RewriteEngine On
RewriteCond %{HTTPS} off
RewriteRule ^(.*)$ https://%{HTTP_HOST}/$1 [R=301,L]

# Security headers (LiteSpeed and Apache both respect mod_headers)
<IfModule mod_headers.c>
  # HSTS — only enable once HTTPS is confirmed stable to avoid lockout
  Header always set Strict-Transport-Security "max-age=31536000; includeSubDomains; preload"
  # Prevent MIME sniffing
  Header always set X-Content-Type-Options "nosniff"
  # Clickjacking protection
  Header always set X-Frame-Options "SAMEORIGIN"
  # Referrer policy (privacy-preserving, still passes origin to same-site)
  Header always set Referrer-Policy "strict-origin-when-cross-origin"
  # Permissions Policy — deny sensitive features by default
  Header always set Permissions-Policy "camera=(), microphone=(), geolocation=(), payment=(self)"
</IfModule>

# If Option A (Node.js proxy) was chosen, uncomment and set the port:
# Note: cPanel Node.js Selector will typically generate its own .htaccess rules
# in the app subfolder. Confirm before enabling these.
#
# RewriteCond %{REQUEST_URI} !^/cgi-bin
# RewriteCond %{REQUEST_URI} !^/.well-known
# RewriteRule ^(.*)$ http://127.0.0.1:<PORT>/$1 [P,L]
```

Notes on HSTS rollout:
- The `preload` directive signals Chrome's HSTS preload list; only submit after 6 months of stable HTTPS and confirmed `max-age` of at least 1 year. Safe to emit the header now; submission to hstspreload.org is a later manual step.
- If SSL cert ever fails at renewal, HSTS locks users out — document in DEPLOY-NOTES.md that cert renewal monitoring is required.

Do NOT enable CSP in this plan. CSP for a Next.js app requires per-page nonces and careful inline-script handling; a naive CSP will break the site. A CSP plan can come later in a v1.1 hardening effort.

`Permissions-Policy: payment=(self)` allows our own PayPal iframe to use Payment Request API (Phase 3 requirement). Denying it by default would break checkout.

| Live search indexing | Crawlers will hit `/admin`, `/api` unless robots.txt is correct |
| Production DB | New surface — checkout writes real order records, not sandbox |
| cPanel .htaccess | File-level config; a mistake here can 500 the entire domain |
